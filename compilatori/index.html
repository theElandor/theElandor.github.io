Esercizio riconoscitori⌗

Scrivere un riconoscitore per stringhe palindrome in C++. Per avere un codice più conciso, si può sfruttare la natura ricorsiva delle stringhe palindrome:

#include <iostream>
#include <cstring>
using namespace std;
bool is_palindrome(char* str, int start, int end)
{
  if(start >= end) return true;
  if(str[start] == str[end] && is_palindrome(str, start+1, end-1))
    return true;
  return false;
}
int main(int argc, char **argv){
  if(argc > 1){
    char* str = argv[1];
    if(is_palindrome(str, 0, strlen(str)-1))
      cout<<"Is Palindrome"<<endl;
    else
      cout<<"Not palindrome"<<endl;
  }
  else
    cout<<"Missing string input"<<endl;

  return 0;
}
